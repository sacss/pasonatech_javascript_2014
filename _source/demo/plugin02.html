プラグイン関連 2 プラグインの使い方2

利用例を別のエフェクトで：jQuery Color

参照：jquery/jquery-color

読み込み

<script src="js/jquery-2.0.3.min.js"></script>
<script src="js/jquery.color-2.1.2.min.js"></script>

Script

jQuery(function($){
	$('button.white_effect01').on('click', function () {
		var $demoBlock = $('.demo__block--type2');
		if(!$demoBlock.hasClass('is-white')) {
			$demoBlock.filter(':not(:animated)').animate({
				backgroundColor: '#f0f'
			}, 1000, function () {
				$(this).addClass('is-white');
			});
		} else {
			$demoBlock.filter(':not(:animated)').animate({
				backgroundColor: '#0f0'
			}, 1000, function () {
				$(this).removeClass('is-white');
			});
		}
		
	});
});

CSS

.demo__block--type2 {
	width: 100%;
	height: 200px;
	background-color: #000;
}

HTML

<div id="demo" class="mb100">
<button>色変更</button>
<div class="demo__block--type2"></div>
