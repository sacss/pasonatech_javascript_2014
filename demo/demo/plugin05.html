エフェクトの緩急 3 コードの変更

コードを、もう少しプログラム的に

Script

jQuery(function($){
	var i = 0;
	var x = 1;
	var size = $('.easing-demo li').length;
	for(x; x < size; x++) {
		$('.easing-demo').before($('#easing1').clone().attr('id', 'easing' + (x + 1)));
	}
	$('button').on('click', function () {
		var mleft = (i % 2)? 0 : '80%';
		$('.easing-demo li').each(function (y) {
			y++;
			$(this).stop().animate({
				marginLeft: mleft
			}, 2000, $('#easing' + y).val());
		});
		i++;
	});
});
